function cycle() {
  ballX += ballXVelocity;
  ballY += ballYVelocity;

  if (ballX + ballRadius > canvas.width) {
    ballXVelocity = -Math.abs(ballXVelocity);
  }
  if (ballX - ballRadius < 0) {
    ballXVelocity = Math.abs(ballXVelocity);
  }

  if (ballY + ballRadius > canvas.height) {
    ballYVelocity = 0;
    ballXVelocity = 0;
  }
  if (ballY - ballRadius < 0) {
    ballYVelocity = Math.abs(ballYVelocity);
  }

  if (ballY)
    if (
      ballX < paddleX + paddleWidth &&
      ballX + ballRadius > paddleX &&
      ballY < paddleY + ballRadius &&
      ballY + paddleHeight > paddleY
    ) {
      if (ballY < paddleY) {
        ballXVelocity = Math.abs(ballXVelocity);
        ballYVelocity = -Math.abs(ballYVelocity);
      }
    }

  //ball 2
  ballX2 -= ballX2Velocity;
  ballY2 -= ballY2Velocity;

  if (ballX2 + ball2Radius > canvas.width) {
    ballX2Velocity = Math.abs(ballX2Velocity);
  }
  if (ballX2 - ball2Radius < 0) {
    ballX2Velocity = -Math.abs(ballX2Velocity);
  }

  if (ballY2 + ball2Radius > canvas.height) {
    ballY2Velocity = 0;
    ballX2Velocity = 0;
  }
  if (ballY2 - ball2Radius < 0) {
    ballY2Velocity = -Math.abs(ballY2Velocity);
  }

  if (
    ballX2 < paddleX + paddleWidth &&
    ballX2 + ball2Radius > paddleX &&
    ballY2 < paddleY + ball2Radius &&
    ballY2 + paddleHeight > paddleY
  ) {
    if (ballY2 < paddleY) {
      ballX2Velocity = -Math.abs(ballX2Velocity);
      ballY2Velocity = Math.abs(ballY2Velocity);
    }
  }

  //ball 3
  ballX3 -= ballX3Velocity;
  ballY3 -= ballY3Velocity;

  //wall collisions
  if (ballX3 + ball3Radius > canvas.width) {
    ballX2Velocity = Math.abs(ballX3Velocity);
  }
  if (ballX3 - ball3Radius < 0) {
    ballX3Velocity = -Math.abs(ballX3Velocity);
  }

  if (ballY3 + ball3Radius > canvas.height) {
    ballY3Velocity = 0;
    ballX3Velocity = 0;
  }
  if (ballY3 - ball3Radius < 0) {
    ballY3Velocity = -Math.abs(ballY3Velocity);
  }

  //paddle collision
  if (
    ballX3 < paddleX + paddleWidth &&
    ballX3 + ball3Radius > paddleX &&
    ballY3 < paddleY + ball3Radius &&
    ballY3 + paddleHeight > paddleY
  ) {
    if (ballY3 < paddleY) {
      ballX3Velocity = -Math.abs(ballX3Velocity);
      ballY3Velocity = Math.abs(ballY3Velocity);
    }
  }

  //ctx.fillStyle="#6699ff"
  //ctx.fillRect(0, 0, canvas.width, canvas.height)

  // Fill with gradient
  ctx.fillStyle = grd;
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  ctx.beginPath();
  ctx.arc(ballX, ballY, 10, 0, 2 * Math.PI);
  ctx.fillStyle = "#ccff99";
  //"#003300";
  ctx.fill();

  ctx.beginPath();
  ctx.arc(ballX2, ballY2, 10, 0, 2 * Math.PI);
  ctx.fillStyle = "#ccff99";
  //"#ffffff";
  ctx.fill();

  ctx.beginPath();
  ctx.arc(ballX3, ballY3, 10, 0, 2 * Math.PI);
  ctx.fillStyle = "#ccff99";
  //"#ffffff";
  ctx.fill();

  if (keys[37]) {
    paddleX -= 4;
  } else if (keys[39]) {
    paddleX += 4;
  }

  if (keys[38]) {
    paddleY -= 4;
  } else if (keys[40]) {
    paddleY += 4;
  }

  ctx.fillStyle = "#ff5050";
  ctx.fillRect(paddleX, paddleY, paddleWidth, paddleHeight);

  if (
    ballY3 + ball3Radius > canvas.height &&
    ballY2 + ball2Radius > canvas.height &&
    ballY + ballRadius > canvas.height
  ) {
    ctx.font = "100px Georgia";
    ctx.fillStyle = "#0048cc";
    ctx.fillText("YOU LOST", 150, canvas.height / 2);

  for (let i = 0; i < ball.length; ++i) {
    //physics
    ball[i].position.x += ball[i].velocity.x;
    ball[i].position.y += ball[i].velocity.y;
    roundB(ball[i].position.x, ball[i].position.y, ball[i].radius);

    if (ball[i].position.x < 0) {
      ball[i].position.x = 0;
      ball[i].velocity.x = Math.abs(ball[i].velocity.x);
    }
    if (ball[i].position.y < 0) {
      ball[i].position.y = 0;
      ball[i].velocity.y = Math.abs(ball[i].velocity.y);
    }
    if (ball[i].position.x > canvas.width) {
      ball[i].position.x = canvas.width;
      ball[i].velocity.x = -Math.abs(ball[i].velocity.x);
    }
    if (ball[i].position.y > canvas.height) {
      ball[i].position.y = canvas.height;
      ball[i].velocity.y = -Math.abs(ball[i].velocity.y);
    }

    if (mouse.down) {
      const dx = ball[i].position.x - mouse.x;
      const dy = ball[i].position.y - mouse.y;
      const d2 = Math.max(dx * dx + dy * dy, 100);
      const mag = 500 / d2 / Math.sqrt(d2);
      ball[i].velocity.x -= mag * dx;
      ball[i].velocity.y -= mag * dy;
    }
  }

  // ctx.rotate(200 * Math.PI / 180);
  // ctx.translate(canvas.width, canvas.height);
  // ctx.beginPath();

  ctx.rotate(-angle);
